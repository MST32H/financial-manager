async function generateMonthlyReport() { try { const { jsPDF } = window.jspdf; const doc = new jsPDF({ orientation: 'portrait', unit: 'mm', format: 'a4' }); // تنظیمات فونت فارسی (استفاده از فونت پیش‌فرض) doc.setFont('helvetica'); // هدر doc.setFillColor(30, 60, 114); doc.rect(0, 0, 210, 30, 'F'); doc.setTextColor(255, 255, 255); doc.setFontSize(20); doc.text('Financial Management Report', 105, 15, { align: 'center' }); doc.setFontSize(10); doc.text(`Generated: ${JalaliDate.today()}`, 105, 23, { align: 'center' }); // خلاصه مالی doc.setTextColor(0, 0, 0); doc.setFontSize(14); doc.text('Financial Summary', 14, 40); const today = JalaliDate.parse(JalaliDate.today()); const monthTx = dm.getMonthTransactions(today.year, today.month); const income = monthTx.filter(t => t.type === 'income').reduce((s,t) => s+t.amount, 0); const expense = monthTx.filter(t => t.type === 'expense').reduce((s,t) => s+t.amount, 0); doc.autoTable({ startY: 45, head: [['Item', 'Amount (Toman)']], body: [ ['Total Balance', dm.getTotalBalance().toLocaleString()], ['Monthly Income', income.toLocaleString()], ['Monthly Expense', expense.toLocaleString()], ['Net Income', (income - expense).toLocaleString()], ['Total Savings', dm.getTotalSavings().toLocaleString()], ['Total Investment', dm.getTotalInvestment().toLocaleString()], ], theme: 'striped', headStyles: { fillColor: [30, 60, 114] } }); // تراکنش‌های اخیر doc.setFontSize(14); doc.text('Recent Transactions', 14, doc.lastAutoTable.finalY + 15); const recentTx = dm.data.transactions.slice(-20).reverse(); if (recentTx.length > 0) { doc.autoTable({ startY: doc.lastAutoTable.finalY + 20, head: [['Date', 'Type', 'Description', 'Amount']], body: recentTx.map(t => [ t.date, t.type === 'income' ? 'Income' : 'Expense', (t.description || '').substring(0, 30), t.amount.toLocaleString() ]), theme: 'grid', headStyles: { fillColor: [16, 185, 129] } }); } // اهداف if (dm.data.savingsGoals.length > 0) { doc.setFontSize(14); doc.text('Savings Goals', 14, doc.lastAutoTable.finalY + 15); doc.autoTable({ startY: doc.lastAutoTable.finalY + 20, head: [['Goal', 'Target', 'Current', 'Progress']], body: dm.data.savingsGoals.map(g => [ g.title, g.targetAmount.toLocaleString(), g.currentAmount.toLocaleString(), g.targetAmount > 0 ? Math.round((g.currentAmount/g.targetAmount)*100) + '%' : '0%' ]), theme: 'grid', headStyles: { fillColor: [245, 158, 11] } }); } // ذخیره doc.save(`financial-report-${JalaliDate.today()}.pdf`); showNotification(' گزارش PDF دانلود شد', 'success'); await dm.addAuditLog('create', 'report', 'گزارش PDF ماهانه تولید شد'); await dm.saveData(); } catch (error) { console.error('PDF generation error:', error); showNotification('❌ خطا در تولید PDF', 'error'); } } window.generateMonthlyReport = generateMonthlyReport;
